// gamma conversion of video RGB to linear RGB for floating point surfaces
// This shader can be run as a screen space pixel shader.
// This shader requires compiling with ps_2_0, but higher is better, see http://en.wikipedia.org/wiki/Pixel_shader to look up what PS version your video card supports.
// If possible, avoid compiling with the software emulation modes (ps_?_sw). Pixel shaders require a lot of processing power to run in real-time software mode.
// This shader will change the gamma curve from the usual video R'G'B' to linear RGB of an image.

sampler s0;

float4 main(float2 tex : TEXCOORD0) : COLOR
{
	float4 s1 = tex2D(s0, tex);// original pixel
	// preserve the sign bits of RGB values
	float4 sb = sign(s1);
	s1 *= sb;
	return sb*pow(s1, 2.4);// linear RGB output
}
