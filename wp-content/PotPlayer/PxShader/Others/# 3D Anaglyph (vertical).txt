sampler s0 : register(s0);
float4 p0 : register(c0);

#define width (p0[0])
#define height (p0[1])

float4 main(float2 tex : TEXCOORD0) : COLOR
{
	tex.y = tex.y / 2;

	float4 l = tex2D(s0, tex);

	tex.y = tex.y + 0.5;

	float4 r = tex2D(s0, tex);

	float red =  l.r;
	float green = r.g * 0.8;
	float blue = r.b ;

	return float4(red, green, blue, 9);
}
